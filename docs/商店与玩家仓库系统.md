# 商店与玩家仓库系统

> 参照《逃离塔科夫》的经济循环：战局内搜刮 → 撤离 → 存入仓库 → 卖给商人 → 购买装备 → 下一局。
> 阵亡则丢失带入战局的装备，但仓库里的东西安全。

---

## 一、经济循环

```
      存入仓库（安全）
      ┌──────┐
      ▼      │
  战利品     │ 下次出战
      │      │ 从仓库取装备
      ▼      │
  ┌────────┐ │
  │  商人  │ │
  │ 可出售 │ │
  │ 可购买 │ │
  └───┬────┘ │
      │      │
      ▼      │
  得到金币 ←─┘
```

**核心规则**：
- 战局内搜刮的物品撤离后进入仓库
- 阵亡则战斗中携带的所有物品丢失（仓库中的不动）
- 仓库中的物品可以卖给商人换金币
- 商人处可以用金币购买装备
- 出战前从仓库选择装备带入战局

## 二、玩家仓库（Stash）

### 2.1 仓库网格

仓库是一个大型网格，存放所有从战局中带出的物品。

```javascript
const STASH_SIZE = { w: 10, h: 10 }; // 10×10 = 100格
// 可扩展（游戏中可购买更大的仓库）
const STASH_UPGRADES = [
  { level: 1, w: 10, h: 10, cost: 0 },
  { level: 2, w: 12, h: 10, cost: 5000 },
  { level: 3, w: 14, h: 12, cost: 15000 },
  { level: 4, w: 16, h: 14, cost: 50000 },
];
```

### 2.2 仓库类

复用 `GridInventory` 类：

```javascript
class PlayerStash {
  constructor() {
    this.level = 1;
    this.size = { ...STASH_UPGRADES[0] };
    this.inventory = new GridInventory(this.size.w, this.size.h);
    this.money = 0;
  }

  // 升级仓库
  upgrade() {
    const next = STASH_UPGRADES[this.level];
    if (!next) return false; // 已满级
    if (this.money < next.cost) return false; // 钱不够
    
    this.money -= next.cost;
    this.level++;
    
    // 创建新网格并迁移物品
    const newInv = new GridInventory(next.w, next.h);
    for (const item of this.inventory.items) {
      const space = newInv.findSpace(item.w, item.h);
      if (space) newInv.placeItem(item, space.x, space.y);
      // 放不下的留在原地（不应发生，新仓库只会更大）
    }
    this.inventory = newInv;
    this.size = next;
    return true;
  }

  // 出战前选择装备（从仓库移到背包）
  equipFromStash(itemId, playerInv) {
    const item = this.inventory.removeItem(itemId);
    if (!item) return false;
    const space = playerInv.findSpace(item.w, item.h);
    if (space) {
      playerInv.placeItem(item, space.x, space.y);
      return true;
    }
    // 背包满了，放回仓库
    this.inventory.placeItem(item, item.x, item.y, item.rotated);
    return false;
  }

  // 战后将背包物品转入仓库
  transferFromRaid(playerInv) {
    for (const item of [...playerInv.items]) {
      const removed = playerInv.removeItem(item.id);
      if (!removed) continue;
      const space = this.inventory.findSpace(removed.w, removed.h);
      if (space) {
        this.inventory.placeItem(removed, space.x, space.y);
      }
      // 仓库满了... 掉落到地上（或提示玩家腾空间）
    }
  }

  // 出售物品
  sellItem(itemId) {
    const item = this.inventory.removeItem(itemId);
    if (!item) return false;
    this.money += item.value || 0;
    return true;
  }
}
```

### 3.2 收购价格计算

```javascript
function getSellPrice(item, vendor) {
  let price = item.value || 0;
  
  // 喜好加成
  if (vendor.likes.some(t => item.tags?.includes(t))) {
    price *= vendor.buyMultiplier;  // 喜好物品给更高价
  }
  // 厌恶减成
  if (vendor.dislikes.some(t => item.tags?.includes(t))) {
    price *= 0.6;  // 不喜欢的物品压价
  }
  
  // 稀有度加成
  if (item.rarity >= 4) price *= 1.5;
  if (item.rarity >= 5) price *= 2.0;
  
  return Math.floor(price);
}
```

### 3.3 购买商品

```javascript
function buyFromVendor(vendorId, itemId, playerStash) {
  const vendor = VENDORS.find(v => v.id === vendorId);
  if (!vendor) return false;
  
  const item = vendor.inventory.find(i => i.id === itemId);
  if (!item) return false;
  if (playerStash.money < item.price) return false;
  
  // 扣钱
  playerStash.money -= item.price;
  
  // 放入仓库
  const space = playerStash.inventory.findSpace(item.w, item.h);
  if (space) {
    playerStash.inventory.placeItem({ ...item, value: Math.floor(item.price * 0.6) }, space.x, space.y);
    return true;
  }
  
  // 仓库满了，退款
  playerStash.money += item.price;
  return false;
}
```

## 四、战前准备（Loadout）

### 4.1 装备流程

```
仓库界面 → 选择装备 → 确认带入战局

带入的物品：
  ├── 头盔
  ├── 护甲
  ├── 胸挂
  ├── 背包 + 背包内物品
  └── 武器

阵亡 → 以上全部丢失
存活撤离 → 全部回到仓库
```

### 4.2 战前准备界面

```javascript
function enterRaid(stash, loadout) {
  // 从仓库移除选中的装备
  const taken = [];
  for (const itemId of loadout.itemIds) {
    const item = stash.inventory.removeItem(itemId);
    if (item) taken.push(item);
  }
  
  // 开局时这些物品赋予玩家
  game.start({
    equipment: taken,
    // ...
  });
}

function extractFromRaid(stash, playerInv) {
  // 存活：背包里的所有物品回仓库
  for (const item of [...playerInv.items]) {
    const removed = playerInv.removeItem(item.id);
    if (!removed) continue;
    const space = stash.inventory.findSpace(removed.w, removed.h);
    if (space) stash.inventory.placeItem(removed, space.x, space.y);
  }
}

function diedInRaid(stash, playerInv) {
  // 阵亡：不做任何事，装备自动消失
  // 玩家只能保留仓库里的东西
}
```

const gameProfile = {
  money: 0,

// 胜利时：物品入仓库
function victoryFunc() {
  const val = calcLootValue();
  game.score += val;
  gameProfile.money += val; // 或者用商人系统卖
  
  // 装备入仓库
  // 头盔、护甲、胸挂、背包、背包内物品
  
  // 存入仓库

}

// 阵亡时：只保留仓库
function gameOverFunc() {
  showRaidSummary({ survived: false, lostValue: calcLootValue() });
  // 所有带入战局的装备丢失
  // 仓库不动

// 出售：从仓库移除，加钱
function sellToVendor(itemId, vendorId) {
  const item = gameProfile.stash.removeItem(itemId);
  if (!item) return false;
  const price = getSellPrice(item, VENDORS.find(v => v.id === vendorId));
  gameProfile.money += price;
  return true;
}

// 全局存档结构
const SAVE_DATA = {

// 序列化仓库（用于存读档）
function serializeStash(stash) {
  return {
    w: stash.width,
    h: stash.height,
    items: stash.items.map(item => ({
      id: item.id,
      name: item.name,
      icon: item.icon,
      w: item.w,
      h: item.h,
      value: item.value,
      rarity: item.rarity,
      x: item.x,
      y: item.y,
      rotated: item.rotated,
      tags: item.tags,
    })),
  };
}

function deserializeStash(data) {
  const inv = new GridInventory(data.w, data.h);
  for (const item of data.items) {
    inv.placeItem(item, item.x, item.y, item.rotated);
  }
  return inv;
}
