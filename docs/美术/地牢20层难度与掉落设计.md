## 五、难度参数

```javascript
// 普通层稀有度权重（最小值 1%）
function getRarityWeights(floor) {
  if (floor <= 4)  return [70, 24,  4,  1,  1];
  if (floor <= 9)  return [20, 38, 30, 10,  2];
  if (floor <= 14) return [ 4, 15, 25, 36, 20];
  if (floor <= 19) return [ 1,  4, 14, 31, 50];
  return              [ 0,  0, 10, 30, 60];
}

// Boss 宝箱稀有度权重（传说不低于 10%）
function getBossChestWeights(floor) {
  if (floor === 5)  return [45, 30, 15, 10];
  if (floor === 10) return [ 0, 35, 35, 30];
  if (floor === 15) return [ 0,  0, 50, 50];
  if (floor === 20) return [ 0,  0, 20, 80];
  return [45, 30, 15, 10];
}

// 获取层数参数
function getFloorParams(floor) {
  const isBossFloor = floor % 5 === 0;
  return {
    enemyHpMult:     1 + (floor - 1) * 0.06,
    enemySpeedMult:  Math.min(1 + (floor - 1) * 0.015, 1.8),
    enemyDmgMult:    1 + (floor - 1) * 0.04,
    enemyCount:      isBossFloor ? 4 + Math.floor(floor/5) : Math.min(6 + Math.floor(floor * 0.7), 22),
    eliteChance:     isBossFloor ? 0 : Math.min(floor * 0.02, 0.5),
    maxRarity:       5,
    exhibitChance:   Math.min(0.04 + floor * 0.012, 0.35),
    isBossFloor:     isBossFloor,
    bossChestCount:  isBossFloor ? Math.floor(floor / 5) + 2 : 0,
  };
}
```
