# 开枪灯光效果设计

> 参考像素风射击游戏的枪口闪光与光照效果。
> 技术方案：Bloom 后处理 + 叠加混合 + 粒子系统。

---

## 一、效果拆解

参考图的开枪效果由三层组成：

```
第一层：枪口闪光（Muzzle Flash）
  白色高亮核心 → 渐变为橙黄 → 外发光
  持续 2~3 帧，瞬间爆发

第二层：弹道光效（Projectile Glow）
  子弹飞行时拖尾残影
  叠加混合（Additive Blending），比背景亮

第三层：击中反馈（Impact Effect）
  命中时散射的黄色火花粒子
  小方块形状，向外扩散，快速衰减
```

---

## 二、实现方案（Canvas 2D）

### 2.1 枪口闪光

```javascript
// 射击时触发
function spawnMuzzleFlash(x, y, angle) {
  const flash = {
    x: x + Math.cos(angle) * 24,  // 枪口位置
    y: y + Math.sin(angle) * 24,
    life: 4,      // 持续 4 帧
    maxLife: 4,
    radius: 14,
    angle: angle,
  };
  muzzleFlashes.push(flash);
}

// 渲染闪光
function renderMuzzleFlashes(ctx) {
  for (const f of muzzleFlashes) {
    const progress = f.life / f.maxLife;  // 1→0
    const alpha = progress;
    const radius = f.radius * (1 + (1 - progress) * 0.5);  // 衰减时稍微扩散
    
    // 外发光（橙色，半透明）
    ctx.save();
    ctx.globalCompositeOperation = 'lighter';  // 叠加混合
    const gradient = ctx.createRadialGradient(f.x, f.y, 0, f.x, f.y, radius);
    gradient.addColorStop(0, `rgba(255, 255, 255, ${alpha})`);     // 白色核心
    gradient.addColorStop(0.3, `rgba(255, 200, 50, ${alpha * 0.8})`);  // 暖黄
    gradient.addColorStop(0.7, `rgba(255, 100, 20, ${alpha * 0.4})`);  // 橙
    gradient.addColorStop(1, `rgba(255, 50, 0, 0)`);                 // 透明边缘
    ctx.fillStyle = gradient;
    ctx.beginPath();
    ctx.arc(f.x, f.y, radius, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
    
    // 衰减
    f.life--;
  }
  // 清除已结束的
  muzzleFlashes = muzzleFlashes.filter(f => f.life > 0);
}
```

### 2.2 子弹拖尾

```javascript
// 子弹数据结构中加入拖尾
function createBullet(x, y, angle) {
  return {
    x, y,
    vx: Math.cos(angle) * 8,
    vy: Math.sin(angle) * 8,
    life: 40,
    trail: [],  // 拖尾轨迹
  };
}

// 更新子弹
function updateBullet(b) {
  // 记录拖尾
  b.trail.push({ x: b.x, y: b.y });
  if (b.trail.length > 6) b.trail.shift();  // 保留最近 6 帧
  
  b.x += b.vx;
  b.y += b.vy;
  b.life--;
}

// 渲染子弹 + 拖尾
function renderBullet(ctx, b) {
  ctx.save();
  ctx.globalCompositeOperation = 'lighter';
  
  // 拖尾（从旧到新，逐渐变亮变粗）
  for (let i = 0; i < b.trail.length; i++) {
    const t = i / b.trail.length;  // 0→1，越新的越亮
    const alpha = t * 0.6;
    const radius = 2 + t * 2;
    ctx.fillStyle = `rgba(255, 200, 50, ${alpha})`;
    ctx.beginPath();
    ctx.arc(b.trail[i].x, b.trail[i].y, radius, 0, Math.PI * 2);
    ctx.fill();
  }
  
  // 子弹本体（白色高亮核心）
  ctx.fillStyle = 'rgba(255, 255, 255, 1)';
  ctx.beginPath();
  ctx.arc(b.x, b.y, 3, 0, Math.PI * 2);
  ctx.fill();
  
  ctx.restore();
}
```

### 2.3 击中火花

```javascript
// 子弹命中时触发
function spawnImpactSpark(x, y) {
  for (let i = 0; i < 8; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = 2 + Math.random() * 4;
    sparks.push({
      x, y,
      vx: Math.cos(angle) * speed,
      vy: Math.sin(angle) * speed,
      life: 10 + Math.random() * 8,
      maxLife: 18,
      size: 2 + Math.random() * 3,
    });
  }
}

// 渲染火花
function renderSparks(ctx) {
  ctx.save();
  ctx.globalCompositeOperation = 'lighter';
  
  for (const s of sparks) {
    const progress = s.life / s.maxLife;
    const alpha = progress;
    ctx.fillStyle = `rgba(255, 200, 50, ${alpha})`;  // 黄橙色
    ctx.fillRect(s.x - s.size / 2, s.y - s.size / 2, s.size, s.size);
    
    s.x += s.vx;
    s.y += s.vy;
    s.vx *= 0.95;  // 减速
    s.vy *= 0.95;
    s.life--;
    s.size *= 0.98;
  }
  
  sparks = sparks.filter(s => s.life > 0);
  ctx.restore();
}
```

### 2.4 整体 Bloom 效果（后处理）

```html
<!-- 方案 A：Canvas 自带滤镜（最简单） -->
<script>
// 在主渲染循环最后调用
function applyBloom(ctx) {
  // 截取当前帧
  const imageData = ctx.getImageData(0, 0, W, H);
  const data = imageData.data;
  
  // 提取亮部并模糊扩散（简化版 bloom）
  // 实际上 Canvas 2D 没有原生 bloom，
  // 可以用 drawImage 叠加一个高斯模糊的副本
  
  // 方法：复制当前画布 → 模糊 → 叠加混合模式画回去
  const tempCanvas = document.createElement('canvas');
  tempCanvas.width = W;
  tempCanvas.height = H;
  const tempCtx = tempCanvas.getContext('2d');
  
  // 复制原画
  tempCtx.drawImage(ctx.canvas, 0, 0);
  
  // 高斯模糊（通过多次缩小放大模拟）
  tempCtx.globalAlpha = 0.3;
  // 水平模糊
  for (let i = 0; i < 3; i++) {
    tempCtx.drawImage(tempCtx.canvas, -2, 0);
    tempCtx.drawImage(tempCtx.canvas, 2, 0);
  }
  
  // 叠加回原画布
  ctx.save();
  ctx.globalCompositeOperation = 'lighter';
  ctx.drawImage(tempCanvas, 0, 0);
  ctx.restore();
}
</script>
```

**方案 B（推荐）：用 CSS filter 做 bloom**

```css
canvas {
  filter: drop-shadow(0 0 4px rgba(255, 200, 50, 0.3));
}
```

叠加混合模式 `lighter` + CSS `drop-shadow` 可以低成本模拟 bloom 效果，性能比 Canvas 后处理好。

## 三、完整接入代码

```javascript
// ====== 玩家射击时 ======
function playerShoot() {
  const angle = player.angle;
  
  // 1. 生成子弹
  const b = createBullet(player.x, player.y, angle);
  player.bullets.push(b);
  
  // 2. 生成枪口闪光
  spawnMuzzleFlash(player.x, player.y, angle);
  
  // 3. 枪口方向额外火花（可选）
  for (let i = 0; i < 3; i++) {
    const spreadAngle = angle + (Math.random() - 0.5) * 0.5;
    const speed = 1 + Math.random() * 2;
    sparks.push({
      x: player.x + Math.cos(angle) * 24,
      y: player.y + Math.sin(angle) * 24,
      vx: Math.cos(spreadAngle) * speed,
      vy: Math.sin(spreadAngle) * speed,
      life: 6,
      maxLife: 6,
      size: 2,
    });
  }
}

// ====== 子弹命中时 ======
function bulletHit(bullet, target) {
  spawnImpactSpark(bullet.x, bullet.y);
  // 命中瞬间额外大闪光
  spawnMuzzleFlash(bullet.x, bullet.y, 0);  // 角度无所谓
}

// ====== 渲染主循环 ======
function render() {
  // ... 渲染地图、角色、敌人等 ...

}
```

## 四、参数调试指南

| 参数 | 参考值 | 效果 |
|------|--------|------|
| 闪光半径 | 10~16px | 越大越夸张 |
| 闪光持续 | 3~5 帧 | 越久越明显 |
| 火花数量 | 6~10 个 | 越多越华丽 |
| 火花大小 | 2~4px | 越大越像素风 |
| 火花速度 | 2~5px/帧 | 越快越有冲击力 |
| 拖尾长度 | 5~8 帧 | 越长弹道越清晰 |
| 叠加模式 | `lighter` | 必须用叠加混合才有发光效果 |
| CSS drop-shadow | 0 0 4px rgba(255,200,50,0.3) | 低成本 bloom |
